001package xyz.mackan.ChatItems.types;
002
003/**
004 * Container class for a pattern to resolve
005 */
006public class ChatPattern {
007        /**
008         * Inventory slots the pattern corresponds to
009         */
010        public enum Type {
011                HAND,
012                OFFHAND,
013                HELMET,
014                CHESTPLATE,
015                LEGS,
016                BOOTS,
017        }
018
019        /**
020         * The type the pattern has
021         */
022        public final Type type;
023
024        /**
025         * The regex pattern that this pattern resolves
026         */
027        public final String pattern;
028
029        /**
030         * Creates a new ChatPattern
031         * @param type {@link ChatPattern.Type} the type of the chatpattern
032         * @param pattern String of the regex pattern
033         */
034        public ChatPattern (Type type, String pattern) {
035                this.type = type;
036                this.pattern = pattern;
037        }
038}
