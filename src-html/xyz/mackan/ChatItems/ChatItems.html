001package xyz.mackan.ChatItems;
002
003import me.pikamug.localelib.LocaleLib;
004import me.pikamug.localelib.LocaleManager;
005import org.bukkit.Bukkit;
006import org.bukkit.configuration.file.FileConfiguration;
007import org.bukkit.plugin.PluginDescriptionFile;
008import xyz.mackan.ChatItems.commands.ChatItemCommand;
009import xyz.mackan.ChatItems.events.PlayerChatEventListener;
010import org.bukkit.plugin.java.JavaPlugin;
011import xyz.mackan.ChatItems.util.ConfigHolder;
012import xyz.mackan.ChatItems.util.PatternManager;
013import xyz.mackan.ChatItems.types.ChatPattern;
014
015import java.util.logging.Level;
016
017/**
018 * Base plugin
019 */
020public class ChatItems extends JavaPlugin {
021        private static LocaleManager localeManager;
022
023        private static PluginDescriptionFile descriptionFile;
024
025        public static LocaleManager getLocaleManager () {
026                return localeManager;
027        }
028
029        public static PluginDescriptionFile getDescriptionFile () { return descriptionFile; }
030
031        public static ConfigHolder configHolder = new ConfigHolder();
032
033        private static ChatItemsAPI api;
034
035        /**
036         * If the ChatItems plugin should handle chat
037         */
038        public static boolean shouldHandleChat = true;
039
040        /**
041         * Sets if ChatItems should handle chat
042         * @param handleChat boolean If ChatItems should handle chat
043         */
044        public static void setHandleChat (boolean handleChat) {
045                shouldHandleChat = handleChat;
046        }
047
048        /**
049         * Loads the configuration for ChatItems from file
050         */
051        private void loadConfig () {
052                FileConfiguration config = this.getConfig();
053
054                configHolder.multiple = config.getBoolean("showQuantities.multiple", true);
055                configHolder.singleItems = config.getBoolean("showQuantities.singleItems", false);
056        }
057
058        /**
059         * Loads ChatItems version support APIs
060         */
061        private void loadAPI () {
062                String packageName = ChatItems.class.getPackage().getName();
063                String internalsName = Bukkit.getServer().getClass().getPackage().getName().split("\\.")[3].substring(1);
064
065                try {
066                        api = (ChatItemsAPI) Class.forName(packageName + ".API.ChatItemsAPI_v" + internalsName).newInstance();
067                        //Bukkit.getServicesManager().register(ChatItemsAPI.class, api, this, ServicePriority.Highest);
068                } catch (ClassNotFoundException | InstantiationException | IllegalAccessException | ClassCastException exception) {
069                        Bukkit.getLogger().log(Level.SEVERE, "ChatItems could not find a valid implementation for this server version.");
070                }
071        }
072
073        /**
074         * Registers the default patterns for ChatItems
075         */
076        private void registerDefaultPatterns () {
077                PatternManager.registerPattern(ChatPattern.Type.HAND, "\\[item\\]");
078                PatternManager.registerPattern(ChatPattern.Type.HAND, "\\[i\\]");
079                PatternManager.registerPattern(ChatPattern.Type.HAND, "\\[hand\\]");
080
081                PatternManager.registerPattern(ChatPattern.Type.OFFHAND, "\\[offhand\\]");
082
083                PatternManager.registerPattern(ChatPattern.Type.HELMET, "\\[helmet\\]");
084                PatternManager.registerPattern(ChatPattern.Type.HELMET, "\\[helm\\]");
085                PatternManager.registerPattern(ChatPattern.Type.HELMET, "\\[head\\]");
086
087
088                PatternManager.registerPattern(ChatPattern.Type.CHESTPLATE, "\\[chestplate\\]");
089                PatternManager.registerPattern(ChatPattern.Type.CHESTPLATE, "\\[chest\\]");
090
091
092                PatternManager.registerPattern(ChatPattern.Type.LEGS, "\\[legs\\]");
093
094                PatternManager.registerPattern(ChatPattern.Type.BOOTS, "\\[boots\\]");
095        }
096
097        @Override
098        public void onEnable() {
099
100                getLogger().info("[ChatItems] enabled.");
101
102                loadAPI();
103
104                this.saveDefaultConfig();
105
106                loadConfig();
107
108                registerDefaultPatterns();
109
110                getServer().getPluginManager().registerEvents(new PlayerChatEventListener(), this);
111
112                LocaleLib localeLib = (LocaleLib) getServer().getPluginManager().getPlugin("LocaleLib");
113                localeManager = localeLib.getLocaleManager();
114
115                this.getCommand("ci").setExecutor(new ChatItemCommand());
116
117                descriptionFile = this.getDescription();
118        }
119
120        @Override
121        public void onDisable() {
122                getLogger().info("[ChatItems] is disabled.");
123        }
124
125        /**
126         * Gets the registered ChatItemsAPI
127         * @return ChatItemsAPI
128         */
129        public static ChatItemsAPI getApi () {
130                return api;
131        }
132}
